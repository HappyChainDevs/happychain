# Getting started with Randomness

import { HomePage } from 'vocs/components'

Getting started with your contract is as easy as:

```solidity
interface IRandomness {
    function random() external view returns (bytes32);
}

contract RandomnessConsumer {
    address public randomnessContractAddress = 0xf4822fC7CB2ec69A5f9D4b5d4a59B949eFfa8311; // Happy Testnet

    //..snip
    function doSomethingWithRandomness() public {
        bytes32 randomValue = IRandomness(randomnessContractAddress).random();
        //..snip
    }
}
```
